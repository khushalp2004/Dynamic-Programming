
///518. Coin Change II
//using dp

typedef unsigned long long ull;
    int change(int amount, vector<int>& coins) {
        vector<ull> dp(amount+1,0);\
        dp[0]=1;

        for(int coin: coins){
            for(int i=coin;i<=amount;i++){
                dp[i]+=dp[i-coin];
            }
        }

        return dp[amount];
    }
