
int sol(int n,vector<int> &nums){
        if(n==1){
            return nums[0];
        }
        if(n==2){
            return max(nums[0],nums[1]);
        }
        
        int prev=nums[0];
        int sprev=0;
        int res=0;

        for(int i=1;i<n;i++){
            res=max(prev,sprev+nums[i]);
            sprev=prev;
            prev=res;
        }
        return res;
    }
    int rob(vector<int>& nums) {
        int n=nums.size();
        return sol(n,nums);
    }
